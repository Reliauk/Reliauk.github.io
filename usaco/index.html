A. Maze Tac Toe

题意简述

奶牛 bessie 在一个迷宫里面走，有些格子是空的，有些格子是禁止进入的，而其它格子分别对应操作 C i j，意思是在一个奶牛井字棋的第 iii 行第 jjj 列放置字母 C∈{M,O}C \in \left\{\texttt{M}, \texttt{O}\right\}C∈{M,O}。当 bessie 到达一个格子，她必须做上述操作。请求出使奶牛井字棋获胜的状态有多少种。

我们说一个奶牛井字棋是获胜的当且仅当存在一行，一列或是一个对角线使得将该部分格子连起来是 MOO\texttt{MOO}MOO 或 OOM\texttt{OOM}OOM。

算法分析

我们只需要模拟 bessie 暴力走并记录状态（三进制装压），判断胜负即可。

时间复杂度为 O(39n2)\text{O}(3 ^ 9 n^2)O(39n2)，其中 n⩽25n \leqslant 25n⩽25，故可以通过。

代码示例

#include <iostream>
#include <cstdio>
#include <set>
#include <algorithm>

using std::cin, std::cout;

const int P = 19683, N = 30, dx[4] = {-1, 0, 1, 0}, dy[4] = {0, -1, 0, 1};

std::set <int> ans;
int n, power[N];
bool vis[N][N][P];
char field[N][N][3];

bool check(int state) {
  int cell[3][3];
  for (int i = 0; i < 3; ++i) {
    for (int j = 0; j < 3; ++j) {
      cell[i][j] = state % 3;
      state /= 3;
    }
  }
  for (int x = 0; x < 3; ++x) {
    if (cell[x][0] == 1 && cell[x][1] == 2 && cell[x][2] == 2) return 1;
    if (cell[x][0] == 2 && cell[x][1] == 2 && cell[x][2] == 1) return 1;
  }
  for (int y = 0; y < 3; ++y) {
    if (cell[0][y] == 1 && cell[1][y] == 2 && cell[2][y] == 2) return 1;
    if (cell[0][y] == 2 && cell[1][y] == 2 && cell[2][y] == 1) return 1;
  }
  if (cell[0][0] == 1 && cell[1][1] == 2 && cell[2][2] == 2) return 1;
  if (cell[0][0] == 2 && cell[1][1] == 2 && cell[2][2] == 1) return 1;
  if (cell[2][0] == 1 && cell[1][1] == 2 && cell[0][2] == 2) return 1;
  if (cell[2][0] == 2 && cell[1][1] == 2 && cell[0][2] == 1) return 1;
  return 0;
}

void dfs(int x, int y, int st) {
  if (vis[x][y][st]) return;
  vis[x][y][st] = 1;
  if (field[x][y][0] == 'M' || field[x][y][0] == 'O') {
    int r = field[x][y][1] - '1', c = field[x][y][2] - '1', node = r * 3 + c;
    int curBit = (st / power[node]) % 3;
    if (!curBit) {
      st += power[node] * (field[x][y][0] == 'M' ? 1 : 2);
      if (!vis[x][y][st] && check(st)) {
        ans.insert(st);
        return;
      }
      vis[x][y][st] = 1;
    }
  }
  for (int i = 0; i < 4; ++i) {
    int nx = x + dx[i], ny = y + dy[i];
    if (nx >= 0 && nx < n && ny >= 0 && ny < n && field[nx][ny][0] != '#') {
      dfs(nx, ny, st);
    }
  }
}

int main() {
  cin >> n;
  int x, y;
  for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++j) {
      cin >> field[i][j][0] >> field[i][j][1] >> field[i][j][2];
      if (field[i][j][0] == 'B') {
        x = i, y = j;
      }
    }
  }
  power[0] = 1;
  for (int i = 1; i <= 9; ++i) power[i] = power[i - 1] * 3;
  dfs(x, y, 0);
  cout << ans.size() << '\n';
  return 0;
}

B. Do You Know Your ABCs?

题意简述

多组数据。

每组数据给你 n(4⩽n⩽7)n(4 \leqslant n \leqslant 7)n(4⩽n⩽7) 个数 x1∼xnx_1 \sim x_nx1​∼xn​ 问有多少组 (A,B,C)(A, B, C)(A,B,C) 使得 ∀v∈x,v∈{A,B,C,A+B,B+C,C+A,A+B+C}\forall v \in x, v\in \left \{A, B, C, A + B, B + C, C + A, A + B + C\right \}∀v∈x,v∈{A,B,C,A+B,B+C,C+A,A+B+C}。

算法分析

我们把 000 加进 xxx 后考虑一个美妙的性质：

∃ v1,v2∈x s.t. v1+v2=A+B+C\exist \ v_1, v_2 \in x \text{ s.t. } v_1 + v_2 = A + B + C ∃ v1​,v2​∈x s.t. v1​+v2​=A+B+C

故我们枚举每一对 v1,v2v_1, v_2v1​,v2​ 就能确定 A+B+CA + B + CA+B+C 的值。不妨设 A⩽B⩽CA \leqslant B \leqslant CA⩽B⩽C：剩下我们需要依次确定 A,BA, BA,B 的值就能得到一组解并判定它。我们找到 xxx 中除 000 的最小值，接下来分两种情况：CCC 和 A+BA + BA+B。分别考虑即可。

代码示例

#include <iostream>
#include <cstdio>
#include <set>
#include <vector>
#include <algorithm>

using std::cin, std::cout;

int T, n;
std::vector <int> x;
std::set <std::multiset <int> > ans;

void check(int a, int b, int c) {
  std::set <int> cur = {0, a, b, c, a + b, a + c, b + c, a + b + c};
  for (auto v: x) {
    if (!cur.count(v)) return;
  }
  ans.insert({a, b, c});
  return ;
}
void calc(int v) {
  std::set <int> cand;
  for (auto c: x) {
    if (c > v) return;
    int minC = std::min(c, v - c);
    if (minC) cand.insert(minC);
  }
  int a = *cand.begin(), b = *(++cand.begin());
  check(v - a - b, a, b);
  check(b - a, a, v - b);
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0), cout.tie(0);
  cin >> T;
  while (T--) {
    cin >> n;
    x.resize(n);
    for (auto &c: x) cin >> c;
    x.emplace_back(0);
    ans.clear();
    for (int i = 0; i < n; ++i) {
      for (int j = i + 1; j <= n; ++j) {
        calc(x[i] + x[j]);
      }
    }
    cout << ans.size() << std::endl;
  }
  return 0;
}

C. Acowdemia

题意简述

bessie 有 n(1⩽n⩽105)n(1 \leqslant n \leqslant 10^5)n(1⩽n⩽105) 个数 c1∼cnc_1 \sim c_nc1​∼cn​。她有 kkk 次机会，每次机会可以将至多 lll 个不同的 ccc 加一。请求出最终她的最大 hhh 指数。

hhh 指数指的是最大的 hhh 满足 ccc 中有至少 hhh 个数 ⩾h\geqslant h⩾h。

算法分析

首先观察到 hhh 指数的单调的，故我们可以二分这个数。

check 需要满足有 hhh 个数 ⩾h\geqslant h⩾h，贪心策略为将 ccc 从大到小排序。

代码示例

#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>

using std::cin, std::cout;

const int N = 1e5 + 5;

int n, k, l;
std::vector <int> c;

bool check(int h) {
  long long sum(0);
  for (int i = 0; i < h; ++i) {
    int v = c[i];
    if (v >= h) continue;
    if (k + v < h) return 0;
    if ((sum += h - v) > 1ll * l * k) return 0;
  }
  return 1;
}

int main() {
  cin >> n >> k >> l;
  c.resize(n);
  for (auto &t: c) cin >> t;
  std::sort(c.begin(), c.end(), std::greater<int>());
  int l = 0, r = n;
  while (l < r) {
    int mid = l + r + 1 >> 1;
    if (check(mid)) l = mid;
    else r = mid - 1;
  }
  cout << l << '\n';
  return 0;
}

A. Just Green Enough

题意简述

给你一个 N×NN \times NN×N（N⩽500N \leqslant 500N⩽500）的方阵，请问存在多少个子矩阵使得子矩阵中最小的元素为 100100100。

算法分析

这题我们可以用容斥做：记 f(x)f(x)f(x) 为使得子矩阵中最小元素 ⩾x\geqslant x⩾x 的子矩阵个数。则答案为 f(100)−f(101)f(100)-f(101)f(100)−f(101)。

想求出 f(x)f(x)f(x)，我们先求出可扩展长度 ext(i,j)\text{ext}(i, j)ext(i,j) 表示 (i,j)(i, j)(i,j) 的右边最多有多少个连续的 ⩾x\geqslant x⩾x 的数。这样，以 (i,j)(i, j)(i,j) 为最上角的合法子矩阵个数就是 ∑k=1i(min⁡m=ki(ext(m,j)))\sum\limits_{k=1}^{i}\left(\min\limits_{m=k}^{i}\left(\text{ext}(m, j)\right)\right)k=1∑i​(m=kmini​(ext(m,j)))。于是我们可以 O(n3)\text{O}(n ^ 3)O(n3) 解决此问题。

代码示例

#include <iostream>
#include <cstdio>
#include <algorithm>

using std::cin, std::cout;
using ll = long long;

const int N = 505;

int n, a[N][N], ext[N][N];

void extendPre(int cand, ll& x) {
  for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j) {
      if (a[i][j] >= cand) {
        ext[i][j] = ext[i][j - 1] + 1;
      } else {
        ext[i][j] = 0;
      }
    }
  }
  for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j) {
      int cur = 1e9 + 7;
      for (int k = i; k; --k) {
        cur = std::min(cur, ext[k][j]);
        x += cur;
      }
    }
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0), cout.tie(0);
  cin >> n;
  for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j) {
      cin >> a[i][j];
    }
  }
  ll ans1 = 0, ans2 = 0;
  extendPre(100, ans1);
  extendPre(101, ans2);
  cout << ans1 - ans2 << '\n';
  return 0;
}

B. Comfortable Cows

题意简述

草地上有 NNN（1≤N≤1051 \leq N \leq 10^51≤N≤105）头奶牛。对于每个 i=1,2,3,…,Ni = 1, 2, 3, \dots, Ni=1,2,3,…,N，请求出当草地上只有前 iii 头奶牛时，需要至少加入多少头奶牛使得没有任何一头奶牛相邻的奶牛恰好为 333 个。

算法分析

我们直接暴力调整即可。

时间复杂度证明：由于每头奶牛只会被最多改一次（奶牛只加不减），故为 O(N)\text{O}(N)O(N)，另外，下标有可能为负数，可以在读入时整体 +1000+1000+1000。

代码示例

#include <iostream>
#include <cstdio>
#include <algorithm>

using std::cin, std::cout;

const int N = 1e5 + 5, F = 5e3 + 5, dx[5] = {0, -1, 0, 1, 0}, dy[5] = {0, 0, -1, 0, 1};

int n, ans;
bool occ[F][F];

bool isComf(int x, int y) {
  if (!occ[x][y]) return 0;
  int near = 0;
  for (int i = 1; i < 5; ++i) {
    if (occ[x + dx[i]][y + dy[i]]) ++near;
  }
  return near == 3;
}
void calc(int x, int y) {
  if (!isComf(x, y)) return;
  for (int i = 1; i < 5; ++i) {
    int nx = x + dx[i], ny= y + dy[i];
    if (!occ[nx][ny]) {
      occ[nx][ny] = 1;
      ++ans;
      for (int j = 0; j < 5; ++j) {
        calc(nx + dx[j], ny + dy[j]);
      }
    }
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0), cout.tie(0);
  cin >> n;
  for (int i = 1; i <= n; ++i) {
    int x, y;
    cin >> x >> y;
    x += 1000, y += 1000;
    if (occ[x][y]) --ans;
    occ[x][y] = 1;
    for (int i = 0; i < 5; ++i) {
      calc(x + dx[i], y + dy[i]);
    }
    cout << ans << '\n';
  }
  return 0;
}

C. Year of the Cow

题意简述

有一个数初始为 000。有 nnn 个数 a1∼ana_1 \sim a_na1​∼an​，请给出一种操作方式使得这 nnn 个数都被遍历且最终回到 000。操作：

  • 花费 111 的代价，x←x−1x \leftarrow x - 1x←x−1。

  • 花费 000 的代价，x←x+12×kx \leftarrow x + 12 \times kx←x+12×k（k∈Zk \in \mathbb{Z}k∈Z）。注意这种操作只能用 KKK 次。

算法分析

由于加减 121212 的倍数不需要花费，所以我们先让 ai←⌊ai12⌋a_i \leftarrow \lfloor \dfrac{a_i}{12}\rfloorai​←⌊12ai​​⌋。这样，我们再把 aia_iai​ 从小到大排序显然是最优的。

代码示例

#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>

using std::cin, std::cout;

const int N = 1e5 + 5;

int n, k, a[N], dif[N];

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0), cout.tie(0);
  cin >> n >> k;
  for (int i = 1; i <= n; ++i) {
    cin >> a[i];
    a[i] = (int)ceil(a[i] / 12.0);
  }
  std::sort(a + 1, a + n + 1);
  for (int i = 1; i <= n; ++i) dif[i] = a[i - 1] - a[i];
  std::sort(dif + 1, dif + n + 1);
  int ans = a[n] * 12;
  for (int i = 1; i < k; ++i) {
    if (!dif[i]) break;
    ans += (dif[i] + 1) * 12;
  }
  cout << ans << '\n';
  return 0;
}
