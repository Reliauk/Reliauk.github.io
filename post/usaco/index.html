A. Maze Tac Toe

题意简述

奶牛 bessie 在一个迷宫里面走，有些格子是空的，有些格子是禁止进入的，而其它格子分别对应操作 C i j，意思是在一个奶牛井字棋的第 iii 行第 jjj 列放置字母 C∈{M,O}C \in \left\{\texttt{M}, \texttt{O}\right\}C∈{M,O}。当 bessie 到达一个格子，她必须做上述操作。请求出使奶牛井字棋获胜的状态有多少种。

我们说一个奶牛井字棋是获胜的当且仅当存在一行，一列或是一个对角线使得将该部分格子连起来是 MOO\texttt{MOO}MOO 或 OOM\texttt{OOM}OOM。

算法分析

我们只需要模拟 bessie 暴力走并记录状态（三进制装压），判断胜负即可。

时间复杂度为 O(39n2)\text{O}(3 ^ 9 n^2)O(39n2)，其中 n⩽25n \leqslant 25n⩽25，故可以通过。

代码示例

#include <iostream>
#include <cstdio>
#include <set>
#include <algorithm>

using std::cin, std::cout;

const int P = 19683, N = 30, dx[4] = {-1, 0, 1, 0}, dy[4] = {0, -1, 0, 1};

std::set <int> ans;
int n, power[N];
bool vis[N][N][P];
char field[N][N][3];

bool check(int state) {
  int cell[3][3];
  for (int i = 0; i < 3; ++i) {
    for (int j = 0; j < 3; ++j) {
      cell[i][j] = state % 3;
      state /= 3;
    }
  }
  for (int x = 0; x < 3; ++x) {
    if (cell[x][0] == 1 && cell[x][1] == 2 && cell[x][2] == 2) return 1;
    if (cell[x][0] == 2 && cell[x][1] == 2 && cell[x][2] == 1) return 1;
  }
  for (int y = 0; y < 3; ++y) {
    if (cell[0][y] == 1 && cell[1][y] == 2 && cell[2][y] == 2) return 1;
    if (cell[0][y] == 2 && cell[1][y] == 2 && cell[2][y] == 1) return 1;
  }
  if (cell[0][0] == 1 && cell[1][1] == 2 && cell[2][2] == 2) return 1;
  if (cell[0][0] == 2 && cell[1][1] == 2 && cell[2][2] == 1) return 1;
  if (cell[2][0] == 1 && cell[1][1] == 2 && cell[0][2] == 2) return 1;
  if (cell[2][0] == 2 && cell[1][1] == 2 && cell[0][2] == 1) return 1;
  return 0;
}

void dfs(int x, int y, int st) {
  if (vis[x][y][st]) return;
  vis[x][y][st] = 1;
  if (field[x][y][0] == 'M' || field[x][y][0] == 'O') {
    int r = field[x][y][1] - '1', c = field[x][y][2] - '1', node = r * 3 + c;
    int curBit = (st / power[node]) % 3;
    if (!curBit) {
      st += power[node] * (field[x][y][0] == 'M' ? 1 : 2);
      if (!vis[x][y][st] && check(st)) {
        ans.insert(st);
        return;
      }
      vis[x][y][st] = 1;
    }
  }
  for (int i = 0; i < 4; ++i) {
    int nx = x + dx[i], ny = y + dy[i];
    if (nx >= 0 && nx < n && ny >= 0 && ny < n && field[nx][ny][0] != '#') {
      dfs(nx, ny, st);
    }
  }
}

int main() {
  cin >> n;
  int x, y;
  for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++j) {
      cin >> field[i][j][0] >> field[i][j][1] >> field[i][j][2];
      if (field[i][j][0] == 'B') {
        x = i, y = j;
      }
    }
  }
  power[0] = 1;
  for (int i = 1; i <= 9; ++i) power[i] = power[i - 1] * 3;
  dfs(x, y, 0);
  cout << ans.size() << '\n';
  return 0;
}

B. Do You Know Your ABCs?

题意简述

多组数据。

每组数据给你 n(4⩽n⩽7)n(4 \leqslant n \leqslant 7)n(4⩽n⩽7) 个数 x1∼xnx_1 \sim x_nx1​∼xn​ 问有多少组 (A,B,C)(A, B, C)(A,B,C) 使得 ∀v∈x,v∈{A,B,C,A+B,B+C,C+A,A+B+C}\forall v \in x, v\in \left \{A, B, C, A + B, B + C, C + A, A + B + C\right \}∀v∈x,v∈{A,B,C,A+B,B+C,C+A,A+B+C}。

算法分析

我们把 000 加进 xxx 后考虑一个美妙的性质：

∃ v1,v2∈x s.t. v1+v2=A+B+C\exist \ v_1, v_2 \in x \text{ s.t. } v_1 + v_2 = A + B + C ∃ v1​,v2​∈x s.t. v1​+v2​=A+B+C

故我们枚举每一对 v1,v2v_1, v_2v1​,v2​ 就能确定 A+B+CA + B + CA+B+C 的值。不妨设 A⩽B⩽CA \leqslant B \leqslant CA⩽B⩽C：剩下我们需要依次确定 A,BA, BA,B 的值就能得到一组解并判定它。我们找到 xxx 中除 000 的最小值，接下来分两种情况：CCC 和 A+BA + BA+B。分别考虑即可。

代码示例

#include <iostream>
#include <cstdio>
#include <set>
#include <vector>
#include <algorithm>

using std::cin, std::cout;

int T, n;
std::vector <int> x;
std::set <std::multiset <int> > ans;

void check(int a, int b, int c) {
  std::set <int> cur = {0, a, b, c, a + b, a + c, b + c, a + b + c};
  for (auto v: x) {
    if (!cur.count(v)) return;
  }
  ans.insert({a, b, c});
  return ;
}
void calc(int v) {
  std::set <int> cand;
  for (auto c: x) {
    if (c > v) return;
    int minC = std::min(c, v - c);
    if (minC) cand.insert(minC);
  }
  int a = *cand.begin(), b = *(++cand.begin());
  check(v - a - b, a, b);
  check(b - a, a, v - b);
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(0), cout.tie(0);
  cin >> T;
  while (T--) {
    cin >> n;
    x.resize(n);
    for (auto &c: x) cin >> c;
    x.emplace_back(0);
    ans.clear();
    for (int i = 0; i < n; ++i) {
      for (int j = i + 1; j <= n; ++j) {
        calc(x[i] + x[j]);
      }
    }
    cout << ans.size() << std::endl;
  }
  return 0;
}

C. Acowdemia

题意简述

bessie 有 n(1⩽n⩽105)n(1 \leqslant n \leqslant 10^5)n(1⩽n⩽105) 个数 c1∼cnc_1 \sim c_nc1​∼cn​。她有 kkk 次机会，每次机会可以将至多 lll 个不同的 ccc 加一。请求出最终她的最大 hhh 指数。

hhh 指数指的是最大的 hhh 满足 ccc 中有至少 hhh 个数 ⩾h\geqslant h⩾h。

算法分析

首先观察到 hhh 指数的单调的，故我们可以二分这个数。

check 需要满足有 hhh 个数 ⩾h\geqslant h⩾h，贪心策略为将 ccc 从大到小排序。

代码示例

#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>

using std::cin, std::cout;

const int N = 1e5 + 5;

int n, k, l;
std::vector <int> c;

bool check(int h) {
  long long sum(0);
  for (int i = 0; i < h; ++i) {
    int v = c[i];
    if (v >= h) continue;
    if (k + v < h) return 0;
    if ((sum += h - v) > 1ll * l * k) return 0;
  }
  return 1;
}

int main() {
  cin >> n >> k >> l;
  c.resize(n);
  for (auto &t: c) cin >> t;
  std::sort(c.begin(), c.end(), std::greater<int>());
  int l = 0, r = n;
  while (l < r) {
    int mid = l + r + 1 >> 1;
    if (check(mid)) l = mid;
    else r = mid - 1;
  }
  cout << l << '\n';
  return 0;
}
